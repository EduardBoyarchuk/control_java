static void playGame() {

        for (Toy toy : toys) {
            if (toy.id == randomNumber) {
                System.out.println("Поздравляем! Вы выиграли " + toy.name + "!");
                toys.remove(toy);
                saveToysToFile();
                return;
            }
        }

    }
